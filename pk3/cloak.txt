Script 534 (Void) //Ghost plasma regen code, much more simple
{
	int plasmaregendelay;
	
	While(1)
	{
		If (plasmaregendelay >= 0 && CheckInventory("PlasmaCount") == 0)
		{
			GiveInventory("PlasmaCell", 1);
			plasmaregendelay = -9;
		}
		plasmaregendelay++;
		Delay(1);
	}
}
